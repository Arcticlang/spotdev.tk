Spot

Take the hassle out of creating a Discord bot.

Get started by creating the example 

import {Spot, Command, CommandListener, EventListener, Event, Message } from "discord-spot";
                        
@CommandListener
@EventListener
class TestBot extends Spot {                   
    constructor() {
        super();
                        
        this.run();
    }     

    @Event("ready")
    onReady(spot: Spot) {
        console.log("Bot is ready.")
    }

    @Command("ping")
    flip(spot: Spot, message: Message, args: string[]) {
        message.reply("Pong!");
    }
}
                        
new TestBot();
